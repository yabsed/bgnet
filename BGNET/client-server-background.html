6 클라이언트-서버 배경지식

세상은 클라이언트-서버로 돌아갑니다. 네트워크에서 일어나는 거의 모든 일은 클라이언트 프로세스와 서버 프로세스가 서로 통신하는 방식으로 이루어집니다. 예를 들어 telnet을 생각해 봅시다. 클라이언트인 telnet으로 원격 호스트의 23번 포트에 접속하면, 그 호스트에서 telnetd라는 프로그램, 즉 서버가 동작을 시작합니다. 이 프로그램은 들어오는 telnet 연결을 처리하고 로그인 프롬프트를 보여주는 등의 일을 합니다.

클라이언트와 서버 사이의 정보 교환은 위 그림에 요약되어 있습니다.

클라이언트-서버 쌍은 SOCK_STREAM, SOCK_DGRAM, 혹은 다른 어떤 방식으로도 통신할 수 있습니다. 중요한 것은 양쪽이 같은 방식으로 통신해야 한다는 점입니다. 대표적인 예로는 telnet/telnetd, ftp/ftpd, Firefox/Apache 같은 조합이 있습니다. 여러분이 ftp를 사용할 때마다 원격지에는 여러분을 위해 응답하는 ftpd라는 프로그램이 존재합니다.

대개 한 머신에는 서버가 하나만 떠 있고, 그 서버가 fork()를 사용해 여러 클라이언트를 처리합니다. 기본 흐름은 이렇습니다. 서버는 연결을 기다렸다가, accept()로 연결을 받아들이고, 그 연결을 처리할 자식 프로세스를 fork()로 생성합니다. 다음 절의 예제 서버가 바로 이 방식을 사용합니다.

6.1 간단한 스트림 서버

이 서버가 하는 일은 스트림 연결을 통해 문자열 “Hello, world!”를 보내는 것뿐입니다. 이 서버를 시험하려면 한 창에서 서버를 실행하고, 다른 창에서 다음과 같이 telnet으로 접속하면 됩니다.

$ telnet remotehostname 3490

여기서 remotehostname은 서버를 실행 중인 머신의 이름입니다.

서버 코드24:

/*
** server.c -- a stream socket server demo
*/

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <errno.h>
#include <string.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <netdb.h>
#include <arpa/inet.h>
#include <sys/wait.h>
#include <signal.h>

#define PORT "3490"  // the port users will be connecting to

#define BACKLOG 10   // how many pending connections queue will hold

void sigchld_handler(int s)
{
    (void)s; // quiet unused variable warning

    // waitpid() might overwrite errno, so we save and restore it:
    int saved_errno = errno;

    while(waitpid(-1, NULL, WNOHANG) > 0);

    errno = saved_errno;
}


// get sockaddr, IPv4 or IPv6:
void *get_in_addr(struct sockaddr *sa)
{
    if (sa->sa_family == AF_INET) {
        return &(((struct sockaddr_in*)sa)->sin_addr);
    }

    return &(((struct sockaddr_in6*)sa)->sin6_addr);
}

int main(void)
{
    // listen on sock_fd, new connection on new_fd
    int sockfd, new_fd;
    struct addrinfo hints, *servinfo, *p;
    struct sockaddr_storage their_addr; // connector's address info
    socklen_t sin_size;
    struct sigaction sa;
    int yes=1;
    char s[INET6_ADDRSTRLEN];
    int rv;

    memset(&hints, 0, sizeof hints);
    hints.ai_family = AF_INET;
    hints.ai_socktype = SOCK_STREAM;
    hints.ai_flags = AI_PASSIVE; // use my IP

    if ((rv = getaddrinfo(NULL, PORT, &hints, &servinfo)) != 0) {
        fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(rv));
        return 1;
    }

    // loop through all the results and bind to the first we can
    for(p = servinfo; p != NULL; p = p->ai_next) {
        if ((sockfd = socket(p->ai_family, p->ai_socktype,
                p->ai_protocol)) == -1) {
            perror("server: socket");
            continue;
        }

        if (setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, &yes,
                sizeof(int)) == -1) {
            perror("setsockopt");
            exit(1);
        }

        if (bind(sockfd, p->ai_addr, p->ai_addrlen) == -1) {
            close(sockfd);
            perror("server: bind");
            continue;
        }

        break;
    }

    freeaddrinfo(servinfo); // all done with this structure

    if (p == NULL)  {
        fprintf(stderr, "server: failed to bind\n");
        exit(1);
    }

    if (listen(sockfd, BACKLOG) == -1) {
        perror("listen");
        exit(1);
    }

    sa.sa_handler = sigchld_handler; // reap all dead processes
    sigemptyset(&sa.sa_mask);
    sa.sa_flags = SA_RESTART;
    if (sigaction(SIGCHLD, &sa, NULL) == -1) {
        perror("sigaction");
        exit(1);
    }

    printf("server: waiting for connections...\n");

    while(1) {  // main accept() loop
        sin_size = sizeof their_addr;
        new_fd = accept(sockfd, (struct sockaddr *)&their_addr,
            &sin_size);
        if (new_fd == -1) {
            perror("accept");
            continue;
        }

        inet_ntop(their_addr.ss_family,
            get_in_addr((struct sockaddr *)&their_addr),
            s, sizeof s);
        printf("server: got connection from %s\n", s);

        if (!fork()) { // this is the child process
            close(sockfd); // child doesn't need the listener
            if (send(new_fd, "Hello, world!", 13, 0) == -1)
                perror("send");
            close(new_fd);
            exit(0);
        }
        close(new_fd);  // parent doesn't need this
    }

    return 0;
}

궁금할 수도 있으니 덧붙이면, 문법적으로 한눈에 보이게 하려고 이 코드를 큰 main() 함수 하나에 넣어 두었습니다. 더 편하다면 작은 함수들로 나눠도 좋습니다.

(또한 sigaction() 부분이 처음 보는 내용일 수도 있는데, 괜찮습니다. 여기 들어 있는 코드는 fork()로 만든 자식 프로세스가 종료될 때 생기는 좀비 프로세스를 회수하는 역할을 합니다. 좀비를 잔뜩 만들고 회수하지 않으면 시스템 관리자가 꽤 예민해질 것입니다.)

이 서버가 보내는 데이터는 다음 절에 나오는 클라이언트를 사용해서 받아볼 수 있습니다.

6.2 간단한 스트림 클라이언트

이쪽은 서버보다 훨씬 더 간단합니다. 이 클라이언트가 하는 일은 명령행에서 지정한 호스트의 3490번 포트에 연결해서 서버가 보내는 문자열을 받아오는 것뿐입니다.

클라이언트 코드25:

/*
** client.c -- a stream socket client demo
*/

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <errno.h>
#include <string.h>
#include <netdb.h>
#include <sys/types.h>
#include <netinet/in.h>
#include <sys/socket.h>

#include <arpa/inet.h>

#define PORT "3490" // the port client will be connecting to 

#define MAXDATASIZE 100 // max number of bytes we can get at once 

// get sockaddr, IPv4 or IPv6:
void *get_in_addr(struct sockaddr *sa)
{
    if (sa->sa_family == AF_INET) {
        return &(((struct sockaddr_in*)sa)->sin_addr);
    }

    return &(((struct sockaddr_in6*)sa)->sin6_addr);
}

int main(int argc, char *argv[])
{
    int sockfd, numbytes;  
    char buf[MAXDATASIZE];
    struct addrinfo hints, *servinfo, *p;
    int rv;
    char s[INET6_ADDRSTRLEN];

    if (argc != 2) {
        fprintf(stderr,"usage: client hostname\n");
        exit(1);
    }

    memset(&hints, 0, sizeof hints);
    hints.ai_family = AF_UNSPEC;
    hints.ai_socktype = SOCK_STREAM;

    if ((rv = getaddrinfo(argv[1], PORT, &hints, &servinfo)) != 0) {
        fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(rv));
        return 1;
    }

    // loop through all the results and connect to the first we can
    for(p = servinfo; p != NULL; p = p->ai_next) {
        if ((sockfd = socket(p->ai_family, p->ai_socktype,
                p->ai_protocol)) == -1) {
            perror("client: socket");
            continue;
        }

        inet_ntop(p->ai_family,
            get_in_addr((struct sockaddr *)p->ai_addr),
            s, sizeof s);
        printf("client: attempting connection to %s\n", s);

        if (connect(sockfd, p->ai_addr, p->ai_addrlen) == -1) {
            perror("client: connect");
            close(sockfd);
            continue;
        }

        break;
    }

    if (p == NULL) {
        fprintf(stderr, "client: failed to connect\n");
        return 2;
    }

    inet_ntop(p->ai_family,
            get_in_addr((struct sockaddr *)p->ai_addr),
            s, sizeof s);
    printf("client: connected to %s\n", s);

    freeaddrinfo(servinfo); // all done with this structure

    if ((numbytes = recv(sockfd, buf, MAXDATASIZE-1, 0)) == -1) {
        perror("recv");
        exit(1);
    }

    buf[numbytes] = '\0';

    printf("client: received '%s'\n",buf);

    close(sockfd);

    return 0;
}

클라이언트를 실행하기 전에 서버를 띄우지 않으면 connect()가 “Connection refused”를 반환한다는 점을 눈여겨보세요. 꽤 유용한 정보입니다.

6.3 데이터그램 소켓

앞에서 sendto()와 recvfrom()를 설명하면서 UDP 데이터그램 소켓의 기본은 이미 다루었으니, 여기서는 예제 프로그램 두 개만 소개하겠습니다. 바로 talker.c와 listener.c입니다.

listener는 어떤 머신에서 실행되며 4950번 포트로 들어오는 패킷을 기다립니다. talker는 지정한 머신의 그 포트로, 사용자가 명령행에 입력한 내용을 담은 패킷을 보냅니다.

데이터그램 소켓은 연결이 없고 성공 여부와 무관하게 패킷을 네트워크로 그냥 쏘아 보내기 때문에, 여기서는 클라이언트와 서버 모두 IPv6만 사용하도록 하겠습니다. 이렇게 하면 서버는 IPv6로 듣고 있는데 클라이언트는 IPv4로 보내는 상황을 피할 수 있습니다. 그런 경우 데이터는 그냥 도착하지 않습니다. (연결형 TCP 스트림 소켓 세계에서는 이런 불일치가 있더라도 한 주소 계열에서 connect()가 실패하면 다른 쪽으로 다시 시도하게 되는 경우가 있습니다.)

다음은 listener.c 소스26입니다:

/*
** listener.c -- a datagram sockets "server" demo
*/

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <errno.h>
#include <string.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <netdb.h>

#define MYPORT "4950"    // the port users will be connecting to

#define MAXBUFLEN 100

// get sockaddr, IPv4 or IPv6:
void *get_in_addr(struct sockaddr *sa)
{
    if (sa->sa_family == AF_INET) {
        return &(((struct sockaddr_in*)sa)->sin_addr);
    }

    return &(((struct sockaddr_in6*)sa)->sin6_addr);
}

int main(void)
{
    int sockfd;
    struct addrinfo hints, *servinfo, *p;
    int rv;
    int numbytes;
    struct sockaddr_storage their_addr;
    char buf[MAXBUFLEN];
    socklen_t addr_len;
    char s[INET6_ADDRSTRLEN];

    memset(&hints, 0, sizeof hints);
    hints.ai_family = AF_INET6; // or set to AF_INET to use IPv4
    hints.ai_socktype = SOCK_DGRAM;
    hints.ai_flags = AI_PASSIVE; // use my IP

    if ((rv = getaddrinfo(NULL, MYPORT, &hints, &servinfo)) != 0) {
        fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(rv));
        return 1;
    }

    // loop through all the results and bind to the first we can
    for(p = servinfo; p != NULL; p = p->ai_next) {
        if ((sockfd = socket(p->ai_family, p->ai_socktype,
                p->ai_protocol)) == -1) {
            perror("listener: socket");
            continue;
        }

        if (bind(sockfd, p->ai_addr, p->ai_addrlen) == -1) {
            close(sockfd);
            perror("listener: bind");
            continue;
        }

        break;
    }

    if (p == NULL) {
        fprintf(stderr, "listener: failed to bind socket\n");
        return 2;
    }

    freeaddrinfo(servinfo);

    printf("listener: waiting to recvfrom...\n");

    addr_len = sizeof their_addr;
    if ((numbytes = recvfrom(sockfd, buf, MAXBUFLEN-1 , 0,
        (struct sockaddr *)&their_addr, &addr_len)) == -1) {
        perror("recvfrom");
        exit(1);
    }

    printf("listener: got packet from %s\n",
        inet_ntop(their_addr.ss_family,
            get_in_addr((struct sockaddr *)&their_addr),
            s, sizeof s));
    printf("listener: packet is %d bytes long\n", numbytes);
    buf[numbytes] = '\0';
    printf("listener: packet contains \"%s\"\n", buf);

    close(sockfd);

    return 0;
}

이제 드디어 getaddrinfo() 호출에서 SOCK_DGRAM을 사용하고 있다는 점을 보세요. 그리고 listen()이나 accept()가 전혀 필요 없다는 점도 주목하세요. 이것이 바로 비연결 데이터그램 소켓의 장점 중 하나입니다.

다음은 talker.c 소스27입니다:

/*
** talker.c -- a datagram "client" demo
*/

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <errno.h>
#include <string.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <netdb.h>

#define SERVERPORT "4950"   // the port users will be connecting to

int main(int argc, char *argv[])
{
    int sockfd;
    struct addrinfo hints, *servinfo, *p;
    int rv;
    int numbytes;

    if (argc != 3) {
        fprintf(stderr,"usage: talker hostname message\n");
        exit(1);
    }

    memset(&hints, 0, sizeof hints);
    hints.ai_family = AF_INET6; // set to AF_INET to use IPv4
    hints.ai_socktype = SOCK_DGRAM;

    rv = getaddrinfo(argv[1], SERVERPORT, &hints, &servinfo);
    if (rv != 0) {
        fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(rv));
        return 1;
    }

    // loop through all the results and make a socket
    for(p = servinfo; p != NULL; p = p->ai_next) {
        if ((sockfd = socket(p->ai_family, p->ai_socktype,
                p->ai_protocol)) == -1) {
            perror("talker: socket");
            continue;
        }

        break;
    }

    if (p == NULL) {
        fprintf(stderr, "talker: failed to create socket\n");
        return 2;
    }

    if ((numbytes = sendto(sockfd, argv[2], strlen(argv[2]), 0,
             p->ai_addr, p->ai_addrlen)) == -1) {
        perror("talker: sendto");
        exit(1);
    }

    freeaddrinfo(servinfo);

    printf("talker: sent %d bytes to %s\n", numbytes, argv[1]);
    close(sockfd);

    return 0;
}

이걸로 전부입니다! 한 머신에서 listener를 실행한 뒤, 다른 머신에서 talker를 실행해 보세요. 둘이 서로 통신하는 모습을 볼 수 있습니다.

이번에는 서버를 꼭 실행할 필요조차 없습니다. talker만 단독으로 실행해도 패킷을 네트워크로 그냥 쏘아 보냅니다. 반대편에서 recvfrom()으로 받을 준비가 된 쪽이 없다면 그 패킷은 그대로 사라집니다. 기억하세요. UDP 데이터그램 소켓으로 보낸 데이터는 도착이 보장되지 않습니다.

단 하나, 이전에도 여러 번 언급했던 작은 디테일이 더 있습니다. 바로 연결된 데이터그램 소켓입니다. 지금이 데이터그램 절이니 여기서 짚고 넘어가겠습니다. 예를 들어 talker가 connect()를 호출해서 listener의 주소를 지정했다고 합시다. 그러면 그 시점부터 talker는 connect()에서 지정한 주소로만 보내고 그 주소로부터만 받을 수 있습니다. 그래서 sendto()와 recvfrom()를 쓸 필요 없이, 단순히 send()와 recv()를 사용하면 됩니다.
